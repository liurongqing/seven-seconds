    从对象中按给定选择器的一组属性取出对应的一组值

    example
    import { get } from 'seven-seconds'
const obj = {
  selector: { to: { val: 'val to select' } },
  target: [1, 2, { a: 'test' }],
}
get(obj, 'selector.to.val', 'target[0]', 'target[2].a');
